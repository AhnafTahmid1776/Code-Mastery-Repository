#include <bits/stdc++.h>

using namespace std;
int a[100];
void printArr(){
   for(int i=0; i<a.length; i++){
    cout << a[i]<< " ";
   }
}

void printPermu(int a[],int cid){
    if(cid== a.length-1){
            printArr(a);
    return 0;

    }
    for(int i=cid; i<a.length;i++){
        swap(a,i,cid);
        printPermu(a,cid+1);
        swap(a,i,cid);
    }

}

int main()
{
    int a[]={1,2,3};
     printPermu(a,0);
}
