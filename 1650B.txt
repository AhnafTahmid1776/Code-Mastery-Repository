#include <bits/stdc++.h>

using namespace std;

int main()

{
    int t;
    cin >> t;
    while(t--){
   long long int l,r,a,x,ans,ans2;
   cin >> l>>r >> a;
   if(a==1){
    cout << r<< endl;
    continue;
   }
   if(r-(r%a)-1 >=l){
    x=r-(r%a)-1;
   }
   else x=r;
   ans=(x/a)+(x%a);
   ans2=(r/a)+(r%a);
   ans=max(ans,ans2);
   cout << ans<< endl;
}
}
