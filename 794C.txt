#include <bits/stdc++.h>

using namespace std;

int main()
{
  long long int f=2e9+10;
  long long int t,n;
  int i;
  cin >> t;
  int a[f];
  while(t--){
    cin >> n;
    for( i=0; i<n;i++){
        cin >>a[i];
    }
    sort(a,a+n);
    if(a[i]<a[i+1] && a[i+2]<a[i+3]){
        cout << "YES";
    for( i=0; i<n;i++){
        cout << a[i]<< endl;
    }
    
    }
    else cout << "NO"<<endl;
    
  }
}
