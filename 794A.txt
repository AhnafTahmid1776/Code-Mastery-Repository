#include <bits/stdc++.h>

using namespace std;

int main()
{

  long long int t,i; cin >> t;
  int a[201];
  while(t--){
    int n;
    cin >> n;
    if(n>=3 && n<=50){
    for(int i=0; i<n;i++){
        cin >>a[i];
    }
    int d= a[0]-a[1];
     sort(a,a+n);
     if(d == a[i]-a[i+1] && n%2==1 &&(a[i]>=0 && a[i]<=100)){
        cout << "YES"<< endl;
     }
     else if(d == a[i]-a[i+1] && n%2==0 && (a[i]>=0 && a[i]<=100)){
        cout << "NO"<< endl;
     }
     else if(d != a[i]-a[i+1]){
            cout << "NO" << endl;
    }
  }
  else cout << "NO" << endl;
  }


}
