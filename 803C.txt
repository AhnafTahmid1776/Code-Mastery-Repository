#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--){
            int n;
             cin >> n;
             int a[n];
             int flag=0;
             int sum=0;
             for (int i=0 ; i < n; i++){
                cin >> a[i];
                sum+=a[i];
             }
             sort(a,a+n);
             for (int i=0 ; i < n-1; i++){
             if(sum==a[i]){
                flag=1;break;
             }
             }
             if (flag==1){
                cout << "Yes"<< endl;
             }
             else cout << "No"<< endl;
    }
}
