#include <bits/stdc++.h>

using namespace std;
char upper(char c){
    return 'A' + (c - 'a');
}
char lower(char c){
    return 'a' + (c - 'A');
}
int main()
{

   string s;
   int c=0,l=0;
   cin >>s ;
   for (int i=0; i<s.size(); i++){
    if (s[i] >= 'A' && s[i] <= 'Z'){
		c++;
	}
	else if (s[i] >= 'a' && s[i] <= 'z'){
		l++;
	}

   }

   for(int i=0; i<s.size(); i++){
    if(l>=c){
            if(s[i] >= 'A' && s[i] <= 'Z'){
                s[i] = lower(s[i]);
            }

   }
   else {

if(s[i] >= 'a' && s[i] <= 'z'){
                s[i] = upper(s[i]);
            }

   }
   }
   cout << s;
   }
