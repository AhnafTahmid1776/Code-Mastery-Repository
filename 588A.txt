#include <bits/stdc++.h>

using namespace std;

int main()

{
    int  n;cin >> n;
   int x=101,sum=0,a[n],p[n];

    for(int i=0; i<n; i++){
        cin >> a[i]>> p[i];

    }
    for(int i=0; i <n; i++){
          x=min(x,p[i]);
      sum+=a[i]*x;
    }



    cout << sum<< endl;
}
