#include<bits/stdc++.h>
using namespace std;
int main()
{
  int t;
  cin >>t;
  long long int i;
  long long int m=2e9+7;
  while(t--){
    int n;
    cin >>n;
    for( i=0;i<m; i++){
        if((n&i)>0 && (n^i)>0){
            break;
        }
        else continue;
    }
    cout << i<<endl;
  }
}
