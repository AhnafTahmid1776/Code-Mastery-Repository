#include <bits/stdc++.h>

using namespace std;

int main()

{
   int t;
   cin  >>t;
   while(t--){
   int n;
   cin >> n;
   int a[n];
   int ct=0;
   for(int i=0; i<n; i++){
        cin >> a[i];
   }
    for(int i=0; i<n; i++ ){
    if((a[i]+a[i+1])%2==1 &&a[i]>0){
    ct++;
    a[i+1]=a[i+2] ;
    }
else if((a[i]+a[i+1])%2==0 &&a[i]>0){
a[i+1]=a[i];
};

   }
   cout << ct<< endl;
}
}
