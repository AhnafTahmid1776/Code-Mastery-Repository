#include <bits/stdc++.h>
using namespace std;
 const int N = int(1e5 + 3);
 int t,n,q;    int a[N];
    long long int c=0,k=0;
int main()
{

    cin >> n;



    for(int i=0; i<n ; i++){
        cin >> t;
        a[t]=i+1;
    }
    int m;
    cin >> m;

    for(int j=0; j<m ; j++){
        cin >> q;
        c+=a[q];
        k+=n-a[q]+1;
    }

   cout << c << " "<< k << endl;


}
