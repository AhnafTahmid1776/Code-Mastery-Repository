#include<bits/stdc++.h>

using namespace std;
typedef long long ll;

int main()
{
    int t;
   cin >>t;
   while(t--){
     ll q1=0;
     ll n,q;
     cin >> n>> q;
     vector<ll> a(n),b(n);
     for(int i=0; i<n; i++){
        cin >>a[i];
     }
     for(int i=n-1; i>=0; i--){
        if(a[i]<=q1){
      b[i]=1;
        }
        else{
            if(q1<q){
                q1++;
                b[i]=1;
            }
            else {b[i]=0;}
        }
     }
     for(int i=0; i<n; i++){
        cout << b[i];
     }
     cout << endl;
   }
}
