#include<bits/stdc++.h>

using namespace std;
void solve(){
int n;
cin >> n;
cout << 2 << endl;
vector<int>vis(n+1,0);
for(int i=1; i<n+1; i++){
  if(vis[i]!=1){
    for(int j=i; j<n;){
        cout << j<< " ";
        vis[j]=1;
        j+=j;
    }
  }
}
cout << endl;
}
int main()
{
    int t;
    cin >>t;
    while(t--){
        solve();
    }
    return 0;
}
