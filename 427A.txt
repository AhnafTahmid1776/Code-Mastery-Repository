#include <iostream>

using namespace std;

int main()
{   
    int n;
    int sum=0, c=0;
    cin >> n;
    int a[n];
    for(int i=0; i<n; i++){
        cin >> a[i];
        sum+=a[i];
        if(sum<0){
            c++;
            sum=0;
        }
    }
    cout << c<< endl;
}
