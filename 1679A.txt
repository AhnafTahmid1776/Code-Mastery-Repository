#include <iostream>

using namespace std;

int main()
{
     int t;
    cin >> t;
    while(t--){
    long long int n;
    cin >> n;
   if(n<4 or (n&1))  {
    cout << "-1"<< endl;
    continue;
   }
   long long int mx,mn;
   mx=n/4;
   mn=n/6;
   if(n%6 ==2 && n-8>=0){
    mn++;
   }
   else if(n%6 ==4 && n-4>=0){
    mn++;
   }

cout << mn<<" "<< mx<< endl;
}
}
