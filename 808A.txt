#include<bits/stdc++.h>

using namespace std;
typedef long long ll;

int main()
{
    int t;
   cin >>t;
   while(t--){
     ll n,flag=0;
    cin >> n;
     vector<ll> a(n);
    for(ll i=0; i<n; i++){
        cin >> a[i];
        if(a[i]%a[0] != 0){
            flag=1;
        }
    }
    if(flag==0)cout << "YES"<< endl;
    else cout << "NO"<< endl;


   }
}
