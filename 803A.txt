#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--){
        int n,k;
        cin >> n>>k;
        int a[n];
        for(int i=0; i<n ; i++){
            cin >> a[i];
        }
        sort(a,a+n);
        int m=a[n-1];
        int sum=0;
        for(int i=0; i<n-1;i++){
            sum+=a[i];
        }
        if(sum==m){
            cout << k<< endl;
        }
        else if(sum>m){
            cout << 0 << endl;
        }
        else cout << 1<< endl;

    }
}
