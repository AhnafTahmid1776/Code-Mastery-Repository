#include <bits/stdc++.h>

using namespace std;

int main()

{
    int n;
    int ct=0;
    cin >> n;
    int a[101];
    for(int i=0; i<n ;i++){
            cin >> a[i];
        if(a[i]== 1){
            ct++;
        }

    }

  if(ct>=1){
            cout << "HARD"<< endl;
        }
        else cout << "EASY"<< endl;

}
