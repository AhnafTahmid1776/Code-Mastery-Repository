#include <bits/stdc++.h>

using namespace std;
typedef long long ll;
#define endl "\n"
#define sz(a) (int)a.size()

const ll mod=1000000007;
const ll N=200005;
int main()
{ ios_base::sync_with_stdio(false);cin.tie(0);cout.tie(0);
    ll t;
    cin >> t;

    while(t--){
    ll n,m;
    cin>>n>>m;
    ll a[n][m];

        for(ll i=0; i<n; i+=2){
            for(ll j=0; j<m; j+=2){
                if((i+j)%4==0){
                    a[i][j]=1;
                a[i+1][j]=0;
                a[i][j+1]=0;
                a[i+1][j+1]=1;

                }
                else {
                    a[i][j]=0;
                a[i+1][j]=1;
                a[i][j+1]=1;
                a[i+1][j+1]=0;
                }

            }

        }

         for(ll i=0; i<n; i++)
    {
        for(ll j=0; j<m; j++)
        {
            cout<<a[i][j]<<" ";
        }
        cout<<endl;
    }

    }
    
}
