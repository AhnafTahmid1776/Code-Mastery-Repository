#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t;
    cin >>t;
    while(t--){
        int n;
        cin >> n;

        int b,a,c;
        a=n/2;

        b=a-1;
        c=n-a-b;
        if(b==c){
            a=a+1;
            c=c-1;
        }
        cout << b<<" " << a<<" "<< c << endl;

    }
}
